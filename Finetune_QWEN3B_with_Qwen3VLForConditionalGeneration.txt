from PIL import Image

def format_data(example):

    image = Image.open(example["image"]).convert("RGB")

    messages = [
        {
            "role": "user",
            "content": [
                {"type": "image", "image": image},
                {"type": "text", "text": example["instruction"]}
            ]
        },
        {
            "role": "assistant",
            "content": [
                {"type": "text", "text": example["output"]}
            ]
        }
    ]
###add_generation_prompt = modle ko answer genartae karna hai aya nhi
    text = processor.apply_chat_template(
        messages,
        tokenize=False,
        add_generation_prompt=False
    )

    return {
        "text": text,
        "image_path": example["image"]
    }
dataset = dataset["train"].map(
    format_data,
    remove_columns=["image", "instruction", "output"]
)

#fine tuning with peft
from peft import LoraConfig, get_peft_model

lora_config = LoraConfig(
    r=16,
    lora_alpha=32,
    target_modules=["q_proj","k_proj","v_proj","o_proj"],
    lora_dropout=0.05,
    bias="none",
    task_type="CAUSAL_LM"
)
###get_peft_model sirf model ko LoRA ke saath prepare karta hai
#Ye weights ko initialize karta hai, lekin abhi backpropagation ya optimizer se train nahi kiya
#Matlab model abhi fine-tuning ke liye ready hai, par trained nahi hai
model = get_peft_model(model, lora_config)

model.print_trainable_parameters()


############ave memory

from PIL import Image

def collate_fn(batch):

    texts = [item["text"] for item in batch]

    images = [
        Image.open(item["image_path"]).convert("RGB").resize((448,448))
        for item in batch
    ]

    # inputs = processor(
    #     text=texts,
    #     images=images,
    #     padding=True,
    #     return_tensors="pt"
    # )
    inputs = processor(
    text=texts,
    images=images,
    padding=True,
    truncation=True,
    max_length=512,
    return_tensors="pt"
)

    inputs["labels"] = inputs["input_ids"]

    return inputs


####final yaha fine tune ho rha hai
from transformers import Trainer

trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=dataset,
    data_collator=collate_fn
)

trainer.train()


#####model save
trainer.save_model("qwen_eob_finetuned")
processor.save_pretrained("qwen_eob_finetuned")

# TEST THE MODEL
image = Image.open("/lambda/nfs/NDS1/EOB/EOB_1/dataset/images/E2102LYE00003001_page_0.png")

messages = [
    {
        "role": "user",
        "content":[
            {"type":"image","image":image},
            {"type":"text","text":"Classify this EOB page"}
        ]
    }
]

text = processor.apply_chat_template(
    messages,
    tokenize=False,
    add_generation_prompt=True
)

inputs = processor(
    text=[text],
    images=[image],
    return_tensors="pt"
).to(model.device)

output = model.generate(**inputs, max_new_tokens=50)

generated_ids = output[:, inputs.input_ids.shape[-1]:]

response = processor.decode(
    generated_ids[0],
    skip_special_tokens=True
)

print(response)
