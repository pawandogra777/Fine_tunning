from PIL import Image

def format_data(example):

    image = Image.open(example["image"]).convert("RGB")

    messages = [
        {
            "role": "user",
            "content": [
                {"type": "image", "image": image},
                {"type": "text", "text": example["instruction"]}
            ]
        },
        {
            "role": "assistant",
            "content": [
                {"type": "text", "text": example["output"]}
            ]
        }
    ]
###add_generation_prompt = modle ko answer genartae karna hai aya nhi
    text = processor.apply_chat_template(
        messages,
        tokenize=False,
        add_generation_prompt=False
    )

    return {
        "text": text,
        "image_path": example["image"]
    }
dataset = dataset["train"].map(
    format_data,
    remove_columns=["image", "instruction", "output"]

from PIL import Image

def collate_fn(batch):

    texts = [item["text"] for item in batch]

    images = [
        Image.open(item["image_path"]).convert("RGB").resize((448,448))
        for item in batch
    ]

    # inputs = processor(
    #     text=texts,
    #     images=images,
    #     padding=True,
    #     return_tensors="pt"
    # )
    inputs = processor(
    text=texts,
    images=images,
    padding=True,
    truncation=True,
    max_length=512,
    return_tensors="pt"
)

    inputs["labels"] = inputs["input_ids"]

    return inputs


####final yaha fine tune ho rha hai

########how many training example
800 pages → depends on tokens per page.

Let’s estimate:

Assume ~500 tokens per page (reasonable for dense text).

800 pages×500 tokens/page=400,000 tokens

If you use sequence length = 512 tokens, then number of training sequences is roughly:

num sequences≈400,000/512≈780

So you only have ~780 training sequences to update 15 million parameters.

780/15,335,424≈0.0000509
Each parameter sees only ~0.00005 sequences per epoch.

Simple Rule (Practical ML Guideline)
Sequences per parameter	Meaning
< 0.01		Very bad (data too small)
0.01 – 0.1	Weak learning
0.1 – 1		Acceptable
1 – 10		Good training
10+		Very strong training

Sequences per parameter		Meaning		Sequences needed	Pages needed (≈)
< 0.01				Very bad	< 153,354		< 153k pages
0.01 – 0.1			Weak learning	153k – 1.53M		153k – 1.53M pages
0.1 – 1				Acceptable	1.53M – 15.3M		1.53M – 15.3M pages
1 – 10				Good training	15.3M – 153M		15.3M – 153M pages
10+				Very strong	153M+			153M+ pages
